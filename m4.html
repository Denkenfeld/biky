async function generateRoute() { if (!userState.userPos) { alert("Warte noch auf GPS-Signal..."); return; } const loadingOverlay = document.getElementById('loading'); loadingOverlay.style.display = 'flex'; const [lon, lat] = userState.userPos; const targetDistKm = userState.baseDist * Math.pow(1.01, userState.level - 1); const radiusKm = targetDistKm / (2 * Math.PI); const radiusDeg = radiusKm / 111.32; let points = []; points.push(`${lon},${lat}`); const numPoints = 6; // Reduziert auf 6 für schnellere Berechnung & Stabilität const randomStartAngle = Math.random() * Math.PI * 2; for (let i = 0; i < numPoints; i++) { const angle = randomStartAngle + (i * (2 * Math.PI / numPoints)); const pLat = lat + Math.sin(angle) * radiusDeg; const pLon = lon + Math.cos(angle) * (radiusDeg / Math.cos(lat * Math.PI / 180)); points.push(`${pLon.toFixed(6)},${pLat.toFixed(6)}`); } points.push(`${lon},${lat}`); const lonlats = points.join('|'); const brouterUrl = `https://brouter.de/brouter?lonlats=${lonlats}&profile=trekking&alternativeidx=0&format=geojson`; // Neuer, stabilerer Proxy-Dienst const proxyUrl = `https://corsproxy.io/?${encodeURIComponent(brouterUrl)}`; try { const response = await fetch(proxyUrl); if (!response.ok) throw new Error("Server antwortet nicht (Code: " + response.status + ")"); const geojson = await response.json(); if (!geojson.features || geojson.features.length === 0) { throw new Error("BRouter konnte keinen Pfad auf Wegen finden."); } renderRoute(geojson); placeCoins(geojson); const actualLen = (geojson.features[0].properties['track-length'] / 1000).toFixed(1); document.getElementById('difficulty-tag').innerText = `${actualLen} km`; } catch (e) { console.error("Detaillierter Fehler:", e); // Fallback: Falls corsproxy.io auch down ist, versuchen wir es mit einer Alternative alert("Routing-Fehler: " + e.message + "\nVersuche es bitte gleich nochmal mit 'GO!'."); } finally { loadingOverlay.style.display = 'none'; } }
